# Calculate average BMI by country
avg_bmi_by_country <- merged_data %>%
  group_by(country) %>%
  summarise(avg_bmi = mean(obs_value, na.rm = TRUE))

# Load world map data
world_map <- map_data("world")

# Merge map data with BMI data
map_data <- merge(world_map, avg_bmi_by_country, by.x = "region", by.y = "country", all.x = TRUE)

# Plotting the map
ggplot() +
  geom_map(data = map_data, map = map_data,
           aes(x = long, y = lat, map_id = region, fill = avg_bmi),
           color = "black", size = 0.2) +
  scale_fill_viridis_c() +
  labs(title = "Average BMI by Country",
       fill = "Average BMI") +
  theme_void()

Bar Chart

  • The bar chart titled “Top 10 Countries by Average Military Expenditure (% of GDP)” compares the percentage of GDP allocated to military spending for ten countries.
# Calculate average Military Expenditure (% of GDP) by country
avg_military_expenditure <- merged_data %>%
  group_by(country) %>%
  summarise(avg_military_expenditure = mean(Military.expenditure....of.GDP., na.rm = TRUE)) %>%
  arrange(desc(avg_military_expenditure))  # Arrange countries by descending average military expenditure

# Select top 10 countries
top_countries <- head(avg_military_expenditure, 10)$country

# Filter top 10 countries
top_10_military_expenditure <- avg_military_expenditure %>%
  filter(country %in% top_countries)

# Plotting the bar chart for top 10 countries
ggplot(top_10_military_expenditure, aes(x = reorder(country, -avg_military_expenditure), y = avg_military_expenditure, fill = country)) +
  geom_bar(stat = "identity", fill = "skyblue") +
  labs(x = "Country", y = "Average Military Expenditure (% of GDP)") +
  ggtitle("Top 10 Countries by Average Military Expenditure (% of GDP)") +
  theme(axis.text.x = element_text(angle = 90, hjust = 1)) +
  scale_fill_discrete(name = "Country")

In the bar chart above:

  • Eritrea: At the top of the chart, we see Eritrea, allocating the highest percentage of its GDP to military spending. This suggests that Eritrea prioritizes defense expenditures significantly.

  • Saudi Arabia: Following Eritrea, Saudi Arabia also dedicates a substantial portion of its GDP to military expenses. The country’s strategic location and regional security concerns likely contribute to this high expenditure.

  • Oman, Jordan, and Israel: These countries fall in the middle range of military spending. Their allocations are notable but not as extreme as Eritrea or Saudi Arabia. Regional conflicts and geopolitical factors often influence their defense budgets.

  • Lebanon, Armenia Republic, and Kuwait: These nations continue the trend of moderate military expenditure. Their percentages are lower than the previous group but still significant. Factors like historical conflicts and regional stability play a role here.

  • Singapore: Known for its strong defense capabilities, Singapore allocates a notable portion of its GDP to military spending. Despite its small size, Singapore maintains a well-equipped armed forces.

  • South Sudan: At the bottom of the chart, we find South Sudan, which spends the least on its military relative to GDP. This could be due to economic challenges and the country’s relatively recent independence.

In summary, this chart highlights the varying priorities and circumstances that lead countries to allocate different percentages of their GDP to military expenses. It reflects geopolitical tensions, regional security concerns, and national defense strategies.

  • The scatter plot titled “BMI vs. GDP per capita” presents a visual correlation between a country’s economic status and the health of its population, as measured by Body Mass Index (BMI). Here’s a detailed breakdown:

  • GDP per Capita: The x-axis represents the GDP per capita in constant 2015 US dollars, ranging from 0 to 50,000. This economic indicator reflects the average income of a country’s citizens and is adjusted for inflation to allow for comparison over time.

  • BMI Percentage: The y-axis measures the BMI percentage, spanning from roughly 10% to over 40%. BMI is a widely used metric to assess whether individuals have a healthy body weight for their height.

  • Data Points: Each black dot on the graph corresponds to a specific pairing of GDP per capita and BMI percentage. The distribution of these points across the graph indicates the relationship between the two variables.

  • Trend Line: A blue line of best fit cuts through the data points, showing an overall upward trend. This suggests that higher GDP per capita is associated with higher BMI percentages.

  • Economic and Health Link: The upward trend may imply that as nations become wealthier, their populations tend to have higher BMI percentages, which could be due to lifestyle changes associated with economic growth.

# Calculate average obs_value by country and sex
avg_obs_value <- merged_data %>%
  group_by(country, sex, current_age) %>%
  summarise(avg_obs_value = mean(obs_value, na.rm = TRUE))

# Identify top 10 countries based on average obs_value
top_countries <- avg_obs_value %>%
  group_by(country) %>%
  summarise(avg_obs_value = mean(avg_obs_value)) %>%
  top_n(10, avg_obs_value) %>%
  pull(country)

# Filter merged_data for top 10 countries
merged_data_top <- merged_data %>%
  filter(country %in% top_countries)

# Plotting time series chart with facets for different sexes
ggplot(merged_data_top, aes(x = year, y = obs_value, color = country)) +
  geom_line() +
  facet_wrap(~ sex, scales = "free_y", ncol = 1) +  # Facet by 'sex' with separate panels
  labs(title = "BMI Over Time by Sex in Top 10 Countries",
       x = "Year", y = "BMI", color = "Country") +
  theme_minimal() +
  theme(legend.position = "top")  # Position legend at the top
